1. 在k8s平台部署java网站项目

1.1 安装jdk、mavn,编译项目

1
2
3
4
5

[root@node01 ~]# yum  -y install git maven java-1.8.0 -y
[root@node01 ~]# git clone https://github.com/lizhenliang/tomcat-java-demo.git
[root@node01 ~]# cd tomcat-java-demo
[root@node01 tomcat-java-demo]# mvn clean package -Dmaven.test.skip=true #mvn编译
[root@node01 tomcat-java-demo]# unzip target/*.war -d target/ROOT

1.2 编写Dockerfile 并打包镜像上传到dockerHub

1
2
3
4
5
6
7
8

[root@node01 tomcat-java-demo]# cat Dockerfile 
FROM tomcat:8.5.61-jdk8-openjdk-slim
RUN rm -rf /usr/local/tomcat/webapps/*
ADD target/ROOT /usr/local/tomcat/webapps/ROOT
[root@node01 tomcat-java-demo] docker build -t java-demo .
[root@node01 tomcat-java-demo] docker login #登录dockerhub官方仓库
[root@node01 tomcat-java-demo] docker tag java-demo hardyxia/java-demo
[root@node01 tomcat-java-demo] docker push  hardyxia/java-demo
  • 推送到镜像仓库,如果是私有仓库,需要配置可信任 vim /etc/docker/deamon.json ,{“insecure”:[“192.168.x.x”]}
  • 将镜像仓库认证凭据保存到k8s Secret 中 
    1

    kubectl create secret docker-registry  registry-auth --docker-username=admin --docker-password=Harbor12345 --docker-server=192.168.x.x
  • 在yml中使用这个认证凭据 
    1
2

        imagePullSecret：
    - name: registry-auth

  [root@node01 tomcat-java-demo] yum install nfs-utils -y   
  #创建nfs服务器及配置,192.168.241.122是服务端，其他为客户端  
  [root@node02 ~] echo "/ifs/kubernetes *(rw,no_root_squash) " >  /etc/exports # *代表所有机器  
  [root@node02 ~] mkdir -p /ifs/kubernetes    
  [root@node02 ~] systemctl start nfs && systemctl enable nfs  
  # 客户端挂载  
  [root@node01 tomcat-java-demo] mount -t nfs 192.168.241.122:/ifs/kubernetes /mnt/

            port: 8080
          initialDelaySeconds: 50
          periodSeconds: 10
        readinessProbe: #就绪健康检查

            path: /
            port: 8080
          initialDelaySeconds: 50
          periodSeconds: 10
        volumeMounts:
        - name: config
          mountPath: "/usr/local/tomcat/webapps/ROOT/WEB-INF/classes/application.yml"
          subPath: "application.yml" #防止覆盖目录
          readOnly: true
        - name: config
          mountPath: "/usr/local/tomcat/webapps/ROOT/WEB-INF/classes/log4j.properties"
          subPath: "log4j.properties" #防止覆盖目
          readOnly: true
      volumes:
      - name: config
        configMap:
          name: java-demo-config
          items: 
          - key: "application.yml"
            path: "application.yml"
          - key: "log4j.properties"
            path: "log4j.properties"

apiVersion: apps/v1
kind: Deployment
metadata:
  name: java-demo-db
  namespace: default
spec:
  selector:
    matchLabels:
      project: www
      app: mysql
  template:
    metadata:
      labels:
        project: www
        app: mysql
    spec:
      containers:
      - image: mysql:5.7.30
        name: db
        resources:
          requests:
            cpu: 0.5
            memory: 300Mi
          limits:
            cpu: 0.5
            memory: 512Mi
        env:
        - name: MYSQL_ROOT_PASSWORD
          valueFrom:
            secretKeyRef:
              name: java-demo-db
              key: mysql-root-password
        - name: MYSQL_PASSWORD
          valueFrom:
            secretKeyRef:
              name: java-demo-db
              key: mysql-password
        - name: MYSQL_USER
          value: "hardy"
        - name: MYSQL_DATABASE
          value: "k8s"
        ports:
        - containerPort: 3306
          name: mysql
        livenessProbe: #存活健康检查
          exec:
            command:
            - sh
            - -c 
            - "mysqladmin ping -u root -p${MYSQL_ROOT_PASSWORD}"
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe: #就绪健康检查
          exec:
            command:
            - sh
            - -c 
            - "mysqladmin ping -u root -p${MYSQL_ROOT_PASSWORD}"
          initialDelaySeconds: 5
          periodSeconds: 10
        volumeMounts:
        - name: data
          mountPath: /var/lib/mysql
      volumes:
      - name: data
        persistentVolumeClaim:
          claimName: java-demo-db
    
---
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: java-demo-db
  namespace: default
spec:
  storageClassName: "managed-nfs-storage"
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 5Gi
    
---
apiVersion: v1
kind: Service
metadata:
  name: java-demo-db
  namespace: default
spec:
  ports:
  - name: mysql
    port: 3306
    targetPort: mysql
  selector:
    project: www
    app: mysql
